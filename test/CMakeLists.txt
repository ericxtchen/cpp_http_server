# Parent project does not export its library target, so this CML implicitly
# depends on being added from it, i.e. the testing is done only from the build
# tree and is not feasible from an install location

project(cpp_http_serverTests LANGUAGES CXX)

# ---- Tests ----

add_executable(cpp_http_server_test source/cpp_http_server_test.cpp)
target_link_libraries(cpp_http_server_test PRIVATE cpp_http_server_lib)
target_compile_features(cpp_http_server_test PRIVATE cxx_std_20)

add_test(NAME cpp_http_server_test COMMAND cpp_http_server_test)

# ---- End-of-file commands ----

add_folders(Test)
